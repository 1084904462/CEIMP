登录界面
	<-
		/login,get

用户登录
	<-
		/userLogin,ajax,post
		{
			accout:String,
			password:String
		}
	->
		model:logStatusBean
		{
			status:String,密码错误、无该用户、登录成功
		}
		//登录成功后添加session
		session:userLogBean
		{
			userId:Long
		}


管理员登录
	<-
		/managerLogin,ajax,post
		{
			account:String,
			password:String
		}
	->
		model:logStatusBean
		{
			status:String,密码错误、无该管理员、登录成功
		}
		//登录成功后添加session
		session:managerLogBean
		{
			managerId:Long
		}

退出登录
	<-
		/loginout,get
