---------------------用户端---------------------
登录界面
	<-
		/login,get

用户登录
	<-
		/userLogin,ajax,post
		LogBean:
		{
			account:String,
			password:String
		}
	->
		StatusBean
		{
			status:String,密码错误、无该用户、登录成功
		}
		//登录成功后添加session
		session:UserLogBean
		{
			userId:Long,
			account:String,
			username:String
		}

管理员登录
	<-
		/managerLogin,ajax,post
		LogBean:
		{
			account:String,
			password:String
		}
	->
		StatusBean
		{
			status:String,密码错误、无该管理员、登录成功
		}
		//登录成功后添加session
		session:ManagerLogBean
		{
			managerId:Long,
			account:String,
			schoolId:Long
		}

用户退出登录
	<-
		/userLoginout,get

管理员退出登录
	<-
		/managerLogout,get

查看奖学金界面
	<-
		/manager/scholarship/{subName},get		//该url用于左侧的导航栏的超链接,grade默认为最新需要综测的年级
		/manager/scholarship/{subName}/{grade},ajax,get	//默认为本次综测的年份
		/manager/scholarship/{subName}/{grade}/{yearScope},ajax,get
	->
		model:ShowScholarshipBean
		{
			scholarshipName:String,	//奖学金名称
			subName:String,	//奖学金名称缩写
			grade:List<String>	//年级，默认降序
			yearScope:Integer	//年份
		}
		model:ScholarshipFormBeanList	//这是个list，需要循环获取
		{
			ScholarshipFormBean
			{
				major:String,		//专业
				classNum:String,	//班级号，这个只有数字，显示的时候需要在前面加上major
				account:String,		//学号
				username:String,	//姓名
				scholarshipName:String	//获奖名称，注意，与上面的scholarshipName，该项可分单项奖学金等
			}
		}

奖学金表格打包
	<-
		/manager/scholarship/zip/{subName},form,post 	//通过js获取勾选列的数据，再模拟表单提交，详情百度
		/manager/scholarship/zip/{subName}/{yearScope},form,post
		ZipInfoBeanList
		{
			ZipInfoBean
			{
				account:String,
				username:String
				scholarshipName:String
			}
		}
